#+BEGIN_QUOTE
Rewrite the INSERTION-SORT procedure to sort into non-increasing
instead of nondecreasing order.
#+END_QUOTE

#+HTML: <section>
#+BEGIN_SRC C
for j = 2 to A.length
  key = A[j]
  i = j - 1
  while i > 0 and A[i] < key
      A[i + 1] = A[i]
      i = i - 1
A[i + 1] = key
#+END_SRC
